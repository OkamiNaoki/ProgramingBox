# 必要はモジュールインポート
import numpy as np
import matplotlib.pyplot as plt

# 計算するアレイのサイズ
size = 512
m = np.arange(-size/2,size/2,1)
Fx,Fy = np.meshgrid(m,m)


# パラメータ設定、単位mm
lam = 600*10**(-6)
dx = 10*10**(-3)
dy = dx

# 光源の距離
d = 200

# 光源の振幅
a0 = 1

# 1個目波面
r0 = np.sqrt((Fx*dx)**2+(Fy*dy)**2+d**2)
u0 = a0*np.exp(1j*2*np.pi*r0/lam)/r0

au = (np.angle(u0))**2
plt.figure()
plt.imshow(au,cmap="gray")
plt.show()


# 2個目光源波面分布
# 1個目からのx軸方向のズレ
k = np.arange(2,100,4)

for x in k:
     r1 = np.sqrt(((Fx-x)*dx)**2+(Fy*dy)**2+d**2)
     u1 = (a0/r1)*np.exp(1j*2*np.pi*r1/lam)
    
    
au1 = np.abs(u1)
pu1 = np.angle(u1)
    
# 干渉計算
u = u0+u1

# 強度画像出力
au = (np.abs(u))**2
plt.figure()
plt.imshow(au,cmap="gray")
plt.show()

#位相分布
atan=np.arctan(2*u.imag/u.real)
plt.figure()
plt.plot(atan)
plt.show()
