import numpy as np
import matplotlib.pyplot as plt

# 
size = 512
m = np.arange(-size/2,size/2,1)
Fx,Fy = np.meshgrid(m,m)


# mm
lam = 600*10**(-6)
dx = 10*10**(-3)
dy = dx

# 
d = 200

# 
a0 = 1

# 1
r0 = np.sqrt((Fx*dx)**2+(Fy*dy)**2+d**2)
u0 = a0*np.exp(1j*2*np.pi*r0/lam)/r0

au = (np.angle(u0))**2
plt.figure()
plt.imshow(au,cmap="gray")
plt.show()


# 2
# 1x
k = np.arange(2,100,4)

for x in k:
     r1 = np.sqrt(((Fx-x)*dx)**2+(Fy*dy)**2+d**2)
     u1 = (a0/r1)*np.exp(1j*2*np.pi*r1/lam)
    
    
au1 = np.abs(u1)
pu1 = np.angle(u1)
    
#

# 
au = (np.abs(u))**2
plt.figure()
